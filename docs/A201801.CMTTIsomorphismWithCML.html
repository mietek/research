mutual
  ↓ₚ : Type → CML.Form
  ↓ₚ (ι P)     = CML.ι P
  ↓ₚ (A ⊃ B)   = ↓ₚ A CML.⊃ ↓ₚ B
  ↓ₚ ([ Ψ ] A) = CML.[ ↓ₚₛ Ψ ] ↓ₚ A

  ↓ₚₛ : ∀ {n} → Types n → List CML.Form
  ↓ₚₛ ∙       = ∙
  ↓ₚₛ (Ξ , A) = ↓ₚₛ Ξ , ↓ₚ A


↓ₐ : ∀ {g} → Assert g
           → CML.Assert
↓ₐ ⟪ Γ ⊫ A ⟫ = CML.⟪ ↓ₚₛ Γ ⊫ ↓ₚ A ⟫


↓ₐₛ : ∀ {d} → {σ : Scopes d}
            → Asserts σ
            → List CML.Assert
↓ₐₛ ∙                = ∙
↓ₐₛ (Δ , ⟪ Γ ⊫ A ⟫) = ↓ₐₛ Δ , ↓ₐ ⟪ Γ ⊫ A ⟫


↓∋ₚ : ∀ {g I A} → {Γ : Types g}
                → Γ ∋⟨ I ⟩ A
                → ↓ₚₛ Γ ∋ ↓ₚ A
↓∋ₚ zero    = zero
↓∋ₚ (suc i) = suc (↓∋ₚ i)


↓∋ₐ : ∀ {d m I A} → {σ : Scopes d} {i : σ ∋⟨ I ⟩ m}
                     {Δ : Asserts σ} {Ψ : Types m}
                  → Δ A201801.AllVec.∋◇⟨ i ⟩ ⟪ Ψ ⊫ A ⟫
                  → ↓ₐₛ Δ ∋ ↓ₐ ⟪ Ψ ⊫ A ⟫
↓∋ₐ zero    = zero
↓∋ₐ (suc i) = suc (↓∋ₐ i)
