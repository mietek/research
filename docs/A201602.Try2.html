data Vec (X : Set) : ℕ → Set where
  ∅   : Vec X zero
  _,_ : ∀{n} → Vec X n → X → Vec X (suc n)

data _∈_ {X : Set} : ∀{n} → X → Vec X n → Set where
  Z  : ∀{n x} {xs : Vec X n}
     → x ∈ (xs , x)
  S_ : ∀{n x y} {xs : Vec X n}
     → x ∈ xs
     → x ∈ (xs , y)

data Ty : Set where
  ⊥   : Ty
  _⊃_ : Ty → Ty → Ty
  _∧_ : Ty → Ty → Ty


Cx : ℕ → Set
Cx = Vec Ty


data _⊢_ {n : ℕ} (Γ : Cx n) : Ty → Set where
  𝑣_     : ∀{A}
         → A ∈ Γ
         → Γ ⊢ A
  𝜆_     : ∀{A B}
         → Γ , A ⊢ B
         → Γ ⊢ A ⊃ B
  _∘_    : ∀{A B}
         → Γ ⊢ A ⊃ B    → Γ ⊢ A
         → Γ ⊢ B
  𝑝⟨_,_⟩ : ∀{A B}
         → Γ ⊢ A        → Γ ⊢ B
         → Γ ⊢ A ∧ B
  𝜋₀_    : ∀{A B}
         → Γ ⊢ A ∧ B
         → Γ ⊢ A
  𝜋₁_    : ∀{A B}
         → Γ ⊢ A ∧ B
         → Γ ⊢ B

⊩_ : Ty → Set
⊩ A = ∀{n} {Γ : Cx n} → Γ ⊢ A


tI : ∀{A} → ⊩ A ⊃ A

tK : ∀{A B} → ⊩ A ⊃ B ⊃ A
tK = 𝜆 𝜆 𝑣 S Z

tS : ∀{A B C} → ⊩ (A ⊃ B ⊃ C) ⊃ (A ⊃ B) ⊃ A ⊃ C
tS = 𝜆 𝜆 𝜆 (𝑣 S S Z ∘ 𝑣 Z) ∘ (𝑣 S Z ∘ 𝑣 Z)


data _⊢_∷_ {n : ℕ} (Γ : Cx n) : (A : Ty) → Γ ⊢ A → Set where
  𝑣²_     : ∀{A}
          → (x : A ∈ Γ)
          → Γ ⊢ A ∷ 𝑣 x
  𝜆²_     : ∀{A B t}
          → Γ , A ⊢ B ∷ t
          → Γ ⊢ A ⊃ B ∷ 𝜆 t
  _∘²_    : ∀{A B t s}
          → Γ ⊢ A ⊃ B ∷ t    → Γ ⊢ A ∷ s
          → Γ ⊢ B ∷ t ∘ s
  𝑝²⟨_,_⟩ : ∀{A B t s}
          → Γ ⊢ A ∷ t        → Γ ⊢ B ∷ s
          → Γ ⊢ A ∧ B ∷ 𝑝⟨ t , s ⟩
  𝜋₀²_    : ∀{A B t}
          → Γ ⊢ A ∧ B ∷ t
          → Γ ⊢ A ∷ 𝜋₀ t
  𝜋₁²_    : ∀{A B t}
          → Γ ⊢ A ∧ B ∷ t
          → Γ ⊢ B ∷ 𝜋₁ t
  !_      : ∀{A t}
          → Γ ⊢ A ∷ t
          → Γ ⊢ A ∷ t

  𝑣³_     : ∀{A}
