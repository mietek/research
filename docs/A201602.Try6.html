data Vec (X : Set) : ℕ → Set where
  ∅   : Vec X zero
  _,_ : ∀{n} → Vec X n → X → Vec X (suc n)

data _∈_ {X : Set} : ∀{n} → X → Vec X n → Set where
  Z  : ∀{n x} {xs : Vec X n}
     → x ∈ (xs , x)
  S_ : ∀{n x y} {xs : Vec X n}
     → x ∈ xs
     → x ∈ (xs , y)


Pr : Set
Pr = ℕ

Cx : ℕ → Set
Cx = Vec Pr

    𝑓_ : ∀{A}
       → A ∈ Δ
       → Δ ∙ Γ ⊢ τ α A

    𝑓²_ : ∀{A}
        → (x : A ∈ Δ)
        → Δ ∙ Γ ⊢ τ α A ∷ 𝑓 x

    𝑓³_ : ∀{A}
        → (x : A ∈ Δ)
        → Δ ∙ Γ ⊢ τ α A ∷ 𝑓 x ∷ 𝑓² x

    𝑣_ : ∀{A}
       → A ∈ Γ
       → Δ ∙ Γ ⊢ τ α A

    𝑣²_ : ∀{A}
        → (x : A ∈ Γ)
        → Δ ∙ Γ ⊢ τ α A ∷ 𝑣 x

    𝑣³_ : ∀{A}
        → (x : A ∈ Γ)
        → Δ ∙ Γ ⊢ τ α A ∷ 𝑣 x ∷ 𝑣² x

    𝜆_ : ∀{A B}
       → Δ ∙ Γ , A ⊢ τ B
       → Δ ∙ Γ ⊢ τ (α A ⊃ B)

    𝜆²_ : ∀{A B t}
        → Δ ∙ Γ , A ⊢ τ B ∷ t
        → Δ ∙ Γ ⊢ τ (α A ⊃ B) ∷ 𝜆 t

    𝜆³_ : ∀{A B t t₂}
        → Δ ∙ Γ , A ⊢ τ B ∷ t ∷ t₂
        → Δ ∙ Γ ⊢ τ (α A ⊃ B) ∷ 𝜆 t ∷ 𝜆² t₂

    _∘_ : ∀{A B}
        → Δ ∙ Γ ⊢ τ (A ⊃ B)    → Δ ∙ Γ ⊢ τ A
        → Δ ∙ Γ ⊢ τ B

    _∘²_ : ∀{A B t s}
         → Δ ∙ Γ ⊢ τ (A ⊃ B) ∷ t    → Δ ∙ Γ ⊢ τ A ∷ s
         → Δ ∙ Γ ⊢ τ B ∷ t ∘ s

    _∘³_ : ∀{A B t t₂ s s₂}
         → Δ ∙ Γ ⊢ τ (A ⊃ B) ∷ t ∷ t₂    → Δ ∙ Γ ⊢ τ A ∷ s ∷ s₂
         → Δ ∙ Γ ⊢ τ B ∷ t ∘ s ∷ t₂ ∘² s₂
