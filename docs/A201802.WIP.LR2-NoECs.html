-- `Val M` says that the term `M` is a value.
data Val {g} : Term g ‚Üí Set
  where
    instance
      VLAM   : ‚àÄ {M} ‚Üí Val (LAM M)
      VTRUE  : Val TRUE
      VFALSE : Val FALSE

-- `_‚Ü¶_` is the CBV computation relation.
infix 3 _‚Ü¶_
data _‚Ü¶_ {g} : Term g ‚Üí Term g ‚Üí Set
  where
    red-APP-LAM  : ‚àÄ {M N} ‚Üí {{_ : Val N}} ‚Üí APP (LAM M) N ‚Ü¶ CUT N M
    red-IF-TRUE  : ‚àÄ {N O} ‚Üí IF TRUE N O ‚Ü¶ N
    red-IF-FALSE : ‚àÄ {N O} ‚Üí IF FALSE N O ‚Ü¶ O


-- Computation is deterministic.
det‚Ü¶ : ‚àÄ {g} ‚Üí {M M‚Ä≤‚ÇÅ M‚Ä≤‚ÇÇ : Term g}
              ‚Üí M ‚Ü¶ M‚Ä≤‚ÇÅ ‚Üí M ‚Ü¶ M‚Ä≤‚ÇÇ
              ‚Üí M‚Ä≤‚ÇÅ ‚â° M‚Ä≤‚ÇÇ
det‚Ü¶ red-APP-LAM  red-APP-LAM  = refl
det‚Ü¶ red-IF-TRUE  red-IF-TRUE  = refl
det‚Ü¶ red-IF-FALSE red-IF-FALSE = refl


-- Computation is type-preserving.
tp‚Ü¶ : ‚àÄ {g M M‚Ä≤ A} ‚Üí {Œì : Types g}
                    ‚Üí M ‚Ü¶ M‚Ä≤ ‚Üí Œì ‚ä¢ M ‚¶Ç A
                    ‚Üí Œì ‚ä¢ M‚Ä≤ ‚¶Ç A
tp‚Ü¶ red-APP-LAM  (app (lam ùíü) ‚Ñ∞) = cut ‚Ñ∞ ùíü
tp‚Ü¶ red-IF-TRUE  (if ùíü ‚Ñ∞ ‚Ñ±)      = ‚Ñ∞
tp‚Ü¶ red-IF-FALSE (if ùíü ‚Ñ∞ ‚Ñ±)      = ‚Ñ±

-- `_‚§á_` is the CBV small-step reduction relation.
infix 3 _‚§á_
data _‚§á_ {g} : Term g ‚Üí Term g ‚Üí Set
  where
    red       : ‚àÄ {M M‚Ä≤} ‚Üí M ‚Ü¶ M‚Ä≤
                         ‚Üí M ‚§á M‚Ä≤
    cong-APP‚ÇÅ : ‚àÄ {M M‚Ä≤ N} ‚Üí M ‚§á M‚Ä≤
                           ‚Üí APP M N ‚§á APP M‚Ä≤ N
    cong-APP‚ÇÇ : ‚àÄ {M N N‚Ä≤} ‚Üí N ‚§á N‚Ä≤
                           ‚Üí APP (LAM M) N ‚§á APP (LAM M) N‚Ä≤
    cong-IF   : ‚àÄ {M M‚Ä≤ N O} ‚Üí M ‚§á M‚Ä≤
                             ‚Üí IF M N O ‚§á IF M‚Ä≤ N O


-- Computation determines small-step reduction.
red-det‚§á : ‚àÄ {g} ‚Üí {M M‚Ä≤‚ÇÅ M‚Ä≤‚ÇÇ : Term g}
                  ‚Üí M ‚Ü¶ M‚Ä≤‚ÇÅ ‚Üí M ‚§á M‚Ä≤‚ÇÇ
                  ‚Üí M‚Ä≤‚ÇÅ ‚â° M‚Ä≤‚ÇÇ
red-det‚§á M‚Ü¶M‚Ä≤‚ÇÅ                   (red M‚Ü¶M‚Ä≤‚ÇÇ) = det‚Ü¶ M‚Ü¶M‚Ä≤‚ÇÅ M‚Ü¶M‚Ä≤‚ÇÇ
red-det‚§á red-APP-LAM              (cong-APP‚ÇÅ (red ()))
red-det‚§á (red-APP-LAM {{VLAM}})   (cong-APP‚ÇÇ (red ()))
red-det‚§á (red-APP-LAM {{VTRUE}})  (cong-APP‚ÇÇ (red ()))
red-det‚§á (red-APP-LAM {{VFALSE}}) (cong-APP‚ÇÇ (red ()))
red-det‚§á red-IF-TRUE              (cong-IF (red ()))
red-det‚§á red-IF-FALSE             (cong-IF (red ()))


-- Small-step reduction is deterministic.
det‚§á : ‚àÄ {g} ‚Üí {M M‚Ä≤‚ÇÅ M‚Ä≤‚ÇÇ : Term g}
              ‚Üí M ‚§á M‚Ä≤‚ÇÅ ‚Üí M ‚§á M‚Ä≤‚ÇÇ
              ‚Üí M‚Ä≤‚ÇÅ ‚â° M‚Ä≤‚ÇÇ
det‚§á (red M‚Ü¶M‚Ä≤‚ÇÅ)         M‚§áM‚Ä≤‚ÇÇ               = red-det‚§á M‚Ü¶M‚Ä≤‚ÇÅ M‚§áM‚Ä≤‚ÇÇ
det‚§á (cong-APP‚ÇÅ M‚§áM‚Ä≤‚ÇÅ)   (red M‚Ü¶M‚Ä≤‚ÇÇ)         = red-det‚§á M‚Ü¶M‚Ä≤‚ÇÇ (cong-APP‚ÇÅ M‚§áM‚Ä≤‚ÇÅ) ‚Åª¬π
det‚§á (cong-APP‚ÇÇ N‚§áN‚Ä≤‚ÇÅ)   (red M‚Ü¶M‚Ä≤‚ÇÇ)         = red-det‚§á M‚Ü¶M‚Ä≤‚ÇÇ (cong-APP‚ÇÇ N‚§áN‚Ä≤‚ÇÅ) ‚Åª¬π
det‚§á (cong-IF M‚§áM‚Ä≤‚ÇÅ)     (red M‚Ü¶M‚Ä≤‚ÇÇ)         = red-det‚§á M‚Ü¶M‚Ä≤‚ÇÇ (cong-IF M‚§áM‚Ä≤‚ÇÅ) ‚Åª¬π
det‚§á (cong-APP‚ÇÅ M‚§áM‚Ä≤‚ÇÅ)   (cong-APP‚ÇÅ M‚§áM‚Ä≤‚ÇÇ)   = (\ M‚Ä≤ ‚Üí APP M‚Ä≤ _) & det‚§á M‚§áM‚Ä≤‚ÇÅ M‚§áM‚Ä≤‚ÇÇ
det‚§á (cong-APP‚ÇÅ (red ())) (cong-APP‚ÇÇ _)
det‚§á (cong-APP‚ÇÇ _)        (cong-APP‚ÇÅ (red ()))
det‚§á (cong-APP‚ÇÇ N‚§áN‚Ä≤‚ÇÅ)   (cong-APP‚ÇÇ N‚§áN‚Ä≤‚ÇÇ)   = (\ N‚Ä≤ ‚Üí APP _ N‚Ä≤) & det‚§á N‚§áN‚Ä≤‚ÇÅ N‚§áN‚Ä≤‚ÇÇ
det‚§á (cong-IF M‚§áM‚Ä≤‚ÇÅ)     (cong-IF M‚§áM‚Ä≤‚ÇÇ)     = (\ M‚Ä≤ ‚Üí IF M‚Ä≤ _ _) & det‚§á M‚§áM‚Ä≤‚ÇÅ M‚§áM‚Ä≤‚ÇÇ


-- Small-step reduction is type-preserving.
tp‚§á : ‚àÄ {g M M‚Ä≤ A} ‚Üí {Œì : Types g}
                    ‚Üí M ‚§á M‚Ä≤ ‚Üí Œì ‚ä¢ M ‚¶Ç A
                    ‚Üí Œì ‚ä¢ M‚Ä≤ ‚¶Ç A
tp‚§á (red M‚Ü¶M‚Ä≤)       ùíü          = tp‚Ü¶ M‚Ü¶M‚Ä≤ ùíü
tp‚§á (cong-APP‚ÇÅ M‚§áM‚Ä≤) (app ùíü ‚Ñ∞)  = app (tp‚§á M‚§áM‚Ä≤ ùíü) ‚Ñ∞
tp‚§á (cong-APP‚ÇÇ M‚§áM‚Ä≤) (app ùíü ‚Ñ∞)  = app ùíü (tp‚§á M‚§áM‚Ä≤ ‚Ñ∞)
tp‚§á (cong-IF M‚§áM‚Ä≤)   (if ùíü ‚Ñ∞ ‚Ñ±) = if (tp‚§á M‚§áM‚Ä≤ ùíü) ‚Ñ∞ ‚Ñ±

-- `_‚§á*_` is the iterated small-step reduction relation.
infix 3 _‚§á*_
data _‚§á*_ {g} : Term g ‚Üí Term g ‚Üí Set
  where
    done : ‚àÄ {M} ‚Üí M ‚§á* M
    step : ‚àÄ {M M‚Ä≤ M‚Ä≥} ‚Üí M ‚§á M‚Ä≥ ‚Üí M‚Ä≥ ‚§á* M‚Ä≤
                       ‚Üí M ‚§á* M‚Ä≤


-- Iterated small-step reduction is transitive.
steps : ‚àÄ {g} ‚Üí {M M‚Ä≤ M‚Ä≥ : Term g}
              ‚Üí M ‚§á* M‚Ä≥ ‚Üí M‚Ä≥ ‚§á* M‚Ä≤
              ‚Üí M ‚§á* M‚Ä≤
steps done                 M‚§á*M‚Ä≤  = M‚§á*M‚Ä≤
steps (step M‚§áM‚Ä¥ M‚Ä¥‚§á*M‚Ä≥) M‚Ä≥‚§á*M‚Ä≤ = step M‚§áM‚Ä¥ (steps M‚Ä¥‚§á*M‚Ä≥ M‚Ä≥‚§á*M‚Ä≤)


-- When reducing down to a value, the initial small step determines the subsequent small steps.
rev-step : ‚àÄ {g} ‚Üí {M M‚Ä≤ M‚Ä≥ : Term g}
                 ‚Üí M ‚§á M‚Ä≥ ‚Üí M ‚§á* M‚Ä≤ ‚Üí {{_ : Val M‚Ä≤}}
                 ‚Üí M‚Ä≥ ‚§á* M‚Ä≤
rev-step M‚§áM‚Ä≥‚ÇÅ   (step M‚§áM‚Ä≥‚ÇÇ M‚Ä≥‚ÇÇ‚§á*M‚Ä≤) with det‚§á M‚§áM‚Ä≥‚ÇÅ M‚§áM‚Ä≥‚ÇÇ
rev-step M‚§áM‚Ä≥    (step _      M‚Ä≥‚§á*M‚Ä≤)  | refl = M‚Ä≥‚§á*M‚Ä≤
rev-step (red ()) done {{VLAM}}
rev-step (red ()) done {{VTRUE}}
rev-step (red ()) done {{VFALSE}}


-- When reducing down to a value, iterated small-step reduction is deterministic.
det‚§á* : ‚àÄ {g} ‚Üí {M M‚Ä≤‚ÇÅ M‚Ä≤‚ÇÇ : Term g}
               ‚Üí M ‚§á* M‚Ä≤‚ÇÅ ‚Üí {{_ : Val M‚Ä≤‚ÇÅ}} ‚Üí M ‚§á* M‚Ä≤‚ÇÇ ‚Üí {{_ : Val M‚Ä≤‚ÇÇ}}
               ‚Üí M‚Ä≤‚ÇÅ ‚â° M‚Ä≤‚ÇÇ
det‚§á* done                    done    = refl
det‚§á* (step M‚§áM‚Ä≥‚ÇÅ M‚Ä≥‚ÇÅ‚§á*M‚Ä≤‚ÇÅ) M‚§á*M‚Ä≤‚ÇÇ = det‚§á* M‚Ä≥‚ÇÅ‚§á*M‚Ä≤‚ÇÅ (rev-step M‚§áM‚Ä≥‚ÇÅ M‚§á*M‚Ä≤‚ÇÇ)
det‚§á* done {{VLAM}}           (step (red ()) _)
det‚§á* done {{VTRUE}}          (step (red ()) _)
det‚§á* done {{VFALSE}}         (step (red ()) _)


-- Iterated small-step reduction is type-preserving.
tp‚§á* : ‚àÄ {g M M‚Ä≤ A} ‚Üí {Œì : Types g}
                     ‚Üí M ‚§á* M‚Ä≤ ‚Üí Œì ‚ä¢ M ‚¶Ç A
                     ‚Üí Œì ‚ä¢ M‚Ä≤ ‚¶Ç A
tp‚§á* done                 ùíü = ùíü
tp‚§á* (step M‚§áM‚Ä≥ M‚Ä≥‚§á*M‚Ä≤) ùíü = tp‚§á* M‚Ä≥‚§á*M‚Ä≤ (tp‚§á M‚§áM‚Ä≥ ùíü)


-- If `M` reduces to `M‚Ä≤`, then `APP M N` reduces to `APP M‚Ä≤ N`.
step-APP‚ÇÅ : ‚àÄ {g} ‚Üí {M M‚Ä≤ N : Term g}
                  ‚Üí M ‚§á* M‚Ä≤
                  ‚Üí APP M N ‚§á* APP M‚Ä≤ N
step-APP‚ÇÅ done                 = done
step-APP‚ÇÅ (step M‚§áM‚Ä≥ M‚Ä≥‚§á*M‚Ä≤) = step (cong-APP‚ÇÅ M‚§áM‚Ä≥) (step-APP‚ÇÅ M‚Ä≥‚§á*M‚Ä≤)


-- If `N` reduces to `N‚Ä≤`, then `APP (LAM M) N` reduces to `APP (LAM M) N‚Ä≤`.
step-APP‚ÇÇ : ‚àÄ {g} ‚Üí {M : Term (suc g)} {N N‚Ä≤ : Term g}
                  ‚Üí N ‚§á* N‚Ä≤
                  ‚Üí APP (LAM M) N ‚§á* APP (LAM M) N‚Ä≤
step-APP‚ÇÇ done                 = done
step-APP‚ÇÇ (step M‚§áM‚Ä≥ M‚Ä≥‚§á*M‚Ä≤) = step (cong-APP‚ÇÇ M‚§áM‚Ä≥) (step-APP‚ÇÇ M‚Ä≥‚§á*M‚Ä≤)


-- If `M` reduces to `M‚Ä≤`, then `IF M N O` reduces to `IF M‚Ä≤ N O`.
step-IF : ‚àÄ {g} ‚Üí {M M‚Ä≤ N O : Term g}
                ‚Üí M ‚§á* M‚Ä≤
                ‚Üí IF M N O ‚§á* IF M‚Ä≤ N O
step-IF done                 = done
step-IF (step M‚§áM‚Ä≥ M‚Ä≥‚§á*M‚Ä≤) = step (cong-IF M‚§áM‚Ä≥) (step-IF M‚Ä≥‚§á*M‚Ä≤)


-- If `M` reduces to `TRUE` and `N` reduces to `N‚Ä≤`, then `IF M N O` reduces to `N‚Ä≤`.
step-IF-TRUE : ‚àÄ {g} ‚Üí {M N N‚Ä≤ O : Term g}
                     ‚Üí M ‚§á* TRUE ‚Üí N ‚§á* N‚Ä≤
                     ‚Üí IF M N O ‚§á* N‚Ä≤
step-IF-TRUE M‚§á*TRUE N‚§á*N‚Ä≤ = steps (step-IF M‚§á*TRUE) (step (red red-IF-TRUE) N‚§á*N‚Ä≤)


-- If `M` reduces to `FALSE` and `O` reduces to `O‚Ä≤`, then `IF M N O` reduces to `O‚Ä≤`.
step-IF-FALSE : ‚àÄ {g} ‚Üí {M N O O‚Ä≤ : Term g}
                      ‚Üí M ‚§á* FALSE ‚Üí O ‚§á* O‚Ä≤
                      ‚Üí IF M N O ‚§á* O‚Ä≤
step-IF-FALSE M‚§á*FALSE N‚§á*N‚Ä≤ = steps (step-IF M‚§á*FALSE) (step (red red-IF-FALSE) N‚§á*N‚Ä≤)

-- `_‚áì_` is the big-step reduction relation.
infix 3 _‚áì_
_‚áì_ : ‚àÄ {g} ‚Üí Term g ‚Üí Term g ‚Üí Set
M ‚áì M‚Ä≤ = M ‚§á* M‚Ä≤ √ó Val M‚Ä≤


-- A big step can be extended with an initial small step.
step‚áì : ‚àÄ {g} ‚Üí {M M‚Ä≤ M‚Ä≥ : Term g}
              ‚Üí M ‚§á M‚Ä≤ ‚Üí M‚Ä≤ ‚áì M‚Ä≥
              ‚Üí M ‚áì M‚Ä≥
step‚áì M‚§áM‚Ä≥ (M‚Ä≥‚§á*M‚Ä≤ , VM‚Ä≤) = step M‚§áM‚Ä≥ M‚Ä≥‚§á*M‚Ä≤ , VM‚Ä≤


-- The initial small step determines the subsequent big step.
rev-step‚áì : ‚àÄ {g} ‚Üí {M M‚Ä≤ M‚Ä≥ : Term g}
                  ‚Üí M ‚§á M‚Ä≤ ‚Üí M ‚áì M‚Ä≥
                  ‚Üí M‚Ä≤ ‚áì M‚Ä≥
rev-step‚áì M‚§áM‚Ä≥ (M‚Ä≥‚§á*M‚Ä≤ , VM‚Ä≤) = rev-step M‚§áM‚Ä≥ M‚Ä≥‚§á*M‚Ä≤ {{VM‚Ä≤}} , VM‚Ä≤


-- Big-step reduction is deterministic.
det‚áì : ‚àÄ {g} ‚Üí {M M‚Ä≤‚ÇÅ M‚Ä≤‚ÇÇ : Term g}
             ‚Üí M ‚áì M‚Ä≤‚ÇÅ ‚Üí M ‚áì M‚Ä≤‚ÇÇ
             ‚Üí M‚Ä≤‚ÇÅ ‚â° M‚Ä≤‚ÇÇ
det‚áì (M‚§á*M‚Ä≤‚ÇÅ , VM‚Ä≤‚ÇÅ) (M‚§á*M‚Ä≤‚ÇÇ , VM‚Ä≤‚ÇÇ) = det‚§á* M‚§á*M‚Ä≤‚ÇÅ {{VM‚Ä≤‚ÇÅ}} M‚§á*M‚Ä≤‚ÇÇ {{VM‚Ä≤‚ÇÇ}}


-- If `M` reduces to `TRUE` and `N` reduces to `N‚Ä≤` then `IF M N O` reduces to `N‚Ä≤`.
leap-IF-TRUE : ‚àÄ {g} ‚Üí {M N N‚Ä≤ O : Term g}
                     ‚Üí M ‚§á* TRUE ‚Üí N ‚áì N‚Ä≤
                     ‚Üí IF M N O ‚áì N‚Ä≤
leap-IF-TRUE M‚§á*TRUE (N‚§á*N‚Ä≤ , VN‚Ä≤) = step-IF-TRUE M‚§á*TRUE N‚§á*N‚Ä≤ , VN‚Ä≤


-- If `M` reduces to `FALSE` and `O` reduces to `O‚Ä≤` then `IF M N O` reduces to `O‚Ä≤`.
leap-IF-FALSE : ‚àÄ {g} ‚Üí {M N O O‚Ä≤ : Term g}
                      ‚Üí M ‚§á* FALSE ‚Üí O ‚áì O‚Ä≤
                      ‚Üí IF M N O ‚áì O‚Ä≤
leap-IF-FALSE M‚§á*FALSE (O‚§á*O‚Ä≤ , VO‚Ä≤) = step-IF-FALSE M‚§á*FALSE O‚§á*O‚Ä≤ , VO‚Ä≤

-- `_‚áì` is the CBV termination relation.
_‚áì : ‚àÄ {g} ‚Üí Term g ‚Üí Set
M ‚áì = Œ£ (Term _) (\ M‚Ä≤ ‚Üí M ‚áì M‚Ä≤)


-- Termination is preserved by small-step reduction.
step‚áì¬∞ : ‚àÄ {g} ‚Üí {M M‚Ä≤ : Term g}
               ‚Üí M ‚§á M‚Ä≤ ‚Üí M‚Ä≤ ‚áì
               ‚Üí M ‚áì
step‚áì¬∞ M‚§áM‚Ä≤ (M‚Ä≥ , M‚Ä≤‚áìM‚Ä≥) = M‚Ä≥ , step‚áì M‚§áM‚Ä≤ M‚Ä≤‚áìM‚Ä≥


-- Termination is preserved by small-step reduction, in reverse.
rev-step‚áì¬∞ : ‚àÄ {g} ‚Üí {M M‚Ä≤ : Term g}
                   ‚Üí M ‚§á M‚Ä≤ ‚Üí M ‚áì
                   ‚Üí M‚Ä≤ ‚áì
rev-step‚áì¬∞ M‚§áM‚Ä≤ (M‚Ä≥ , M‚Ä≤‚áìM‚Ä≥) = M‚Ä≥ , rev-step‚áì M‚§áM‚Ä≤ M‚Ä≤‚áìM‚Ä≥


-- If `M` reduces to `TRUE` and `N` terminates, then `IF M N O` terminates.

                     ‚Üí M ‚§á* TRUE ‚Üí N ‚áì
                     ‚Üí IF M N O ‚áì
halt-IF-TRUE M‚§á*TRUE (N‚Ä≤ , N‚áìN‚Ä≤) = N‚Ä≤ , leap-IF-TRUE M‚§á*TRUE N‚áìN‚Ä≤


-- If `M` reduces to `FALSE` and `O` terminates, then `IF M N O` terminates.
halt-IF-FALSE : ‚àÄ {g} ‚Üí {M N O : Term g}
                      ‚Üí M ‚§á* FALSE ‚Üí O ‚áì
                      ‚Üí IF M N O ‚áì
halt-IF-FALSE M‚§á*FALSE (O‚Ä≤ , O‚áìO‚Ä≤) = O‚Ä≤ , leap-IF-FALSE M‚§á*FALSE O‚áìO‚Ä≤
